FROM employees

SELECT* emp_name, department_id

-- 사칙연산 사용가능
SELECT emp_name
        , salary / 30            as 일당
        , salary                 as 월급
        , salary - salary * 0.1  as 실수령액
        , salary * 12            as 연봉
FROM employees
ORDER BY 3 DESC; -- 월급  내림차순 정렬
